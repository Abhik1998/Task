let actorDefinition id = 
    actor {
        body (
          let rec loop() =
              messageHandler {
                  let! msg = Message.receive None
                  do printfn "%s handled %s" id msg
                  return! loop()
              }
          loop()
        )
    }

let actors = 
    [1..5] 
    |> List.map (fun i -> actorDefinition (string i) |> Actor.start) 
    |> Routing.roundRobin

    actors <-- "Msg " + (string i)
